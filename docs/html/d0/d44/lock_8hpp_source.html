33/**
34 * \brief A lock that keeps a histogram of how frequently each value is present
35 *
36 * \todo refactor the histogram into another class, and have it a separate
37 * dependency from the lock
38 *
39 * @tparam max_elem The largest value on the lock
40 * @tparam min_elem The smallest value on the lock
41 * @tparam T The type of the values on the lock
42 */
43template<std::size_t max_elem,
44 std::size_t min_elem = 0,
45 std::signed_integral T = std::int32_t>
46 requires(max_elem > min_elem)
47class histogram_lock {
48 public:
49 /**
50 * \brief Construct a new lock object at position init
51 *
52 * \param init Value that the lock starts at
53 */
54 histogram_lock(T init) { set_current_value(init); }
55
56 /**
57 * \brief sets the current position of the lock
58 *
59 * \param val New position
60 * \throws std::range_error Value not between min and max
61 */
62 void set_current_value(T val) {
63 bounds_check(val);
64 current_value = std::move(val);
65 histogram[val - min]++;
66 }
67
68 /**
69 * \brief Sets the current position of the lock
70 *
71 * \param val New position
72 * \throws std::range_error Value not between min and max
73 */
74 void operator==(T val) { set_current_value(val); }
75
76 /**
77 * \brief Increases the position of the lock by \ref val "val" amount
78 *
79 * \param val Delta
80 * \throws std::range_error Value not between min and max
81 */
82
83 void operator+=(T val) {
84 // Add val to current_value, then wrap and set
85 set_current_value(wrap_to_range(current_value + val));
86 }
87
88 /**
89 * \brief Decreases the position of the lock by \ref val "val" amount
90 *
91 * \param val Delta
92 * \throws std::range_error Value not between min and max
93 */
94 void operator-=(T val) { this->operator+=(-val); }
95
96 /**
97 * \brief Get the current value object
98 */
99 const T get_current_value() const { return current_value; }
100
101 /**
102 * \brief Gets how many times the lock was at val
103 *
104 * \param val
105 * \return count
106 */
107 T get_frequency(T val) {
108 bounds_check(val);
109 return histogram[val - min];
110 }
111
112 private:
113 static constexpr T min = static_cast<T>(min_elem);
114 static constexpr T max = static_cast<T>(max_elem);
115 static constexpr T range = max - min + 1;
116
117 std::array<T, max - min + 1> histogram{};
118 T current_value;
119
120 /**
121 * \brief Checks if the argument is in the range that this lock can handle
122 *
123 * Throws if out of bounds or returns true (can not return false).
124 * \param t checked value
125 * \return true
126 * \throws std::range_error value not in range
127 */
128 bool bounds_check(T t) {
129 if (t > max || t < min)
130 throw std::range_error{ std::format(
131 "Value {} not in lock bounds [{} - {}]", t, min, max) };
132 return true;
133 }
134
135 /**
136 * \brief Does the modular wraping math.
137 */
138 static constexpr T wrap_to_range(T val) {
139 if (val >= min && val <= max) [[likely]]
140 return val;
141 return min + ((val - min) % range + range) % range;
142 }
143};
